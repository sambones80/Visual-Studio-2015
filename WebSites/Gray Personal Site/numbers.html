Fizz Buzz

Upon entering two numbers, this tool will use JavaScript to calculate the multiples of each number up to 100.

Fizz Buzz

Enter two numbers




Fizz Buzz JS

    // Initialize Value
    $('#calculateFizzBuzz').click(calcFizzBuzz);

    function calcFizzBuzz() {

        var fbNum1 = 0;
        var fbNum2 = 0;

    // Clear Previous Answers
        $('#fizzBuzz100').html('')
        $('#fizzBuzzOutput').html('')

    // Print 1 - 100
        for (var i = 1; i <= 100; i++) {
            $('#fizzBuzz100').append('
' + i);
        }

    // Parse String Number Data Into Number Data
        var fbNum1 = parseInt($('#fizzBuzz1').val());
        var fbNum2 = parseInt($('#fizzBuzz2').val());

    // Calculate Fizz Buzz
        for (var i = 1; i <= 100; i++) {
            if (i % fbNum1 !== 0 && i % fbNum2 !== 0)

                // If modulus returns a remainder for both user numbers - print 1-100 number
                $('#fizzBuzzOutput').append('
' + i);
            else if (i % fbNum1 === 0 && i % fbNum2 === 0)

                // If modulus returns NO remainder for both user numbers - print "Fizz Buzz"
                $('#fizzBuzzOutput').append('
' + 'Fizz Buzz');
            else if (i % fbNum1 === 0)

                // If modulus returns NO remainder for first user number - print "Fizz"
                $('#fizzBuzzOutput').append('
' + 'Fizz');
            else if (i % fbNum2 === 0)

                // If modulus returns NO remainder for second user number - print "Buzz"
                $('#fizzBuzzOutput').append('
' + 'Buzz');
        }
    }

        var reverseWord = splitWord.reverse();

Palindrome Checker

This is a C# console program that will accept a word and determine if it is a palindrome.

Palindrome Checker C# Code

namespace Palindrome
{
    class Program
    {
        static void Main(string[] args)
        {
            // Initialize the value
            string palWord = "";

            // Ask user to input a word
            Console.WriteLine("--- Palindrome Checker ---");
            Console.WriteLine("");
            Console.WriteLine("This program will accept a word and determine if it is a palindrome.");
            Console.WriteLine("");
            Console.WriteLine("Enter a word to see if it's a palindrome:");

            palWord = Console.ReadLine();

            // Break string apart to array of characters
            char[] palArray = palWord.ToCharArray();

            // Reverse the order of characters in the array
            Array.Reverse(palArray);

            // Create new string from the reversed array
            string revWord = new string(palArray);

            // Compare user's original word to the reversed word and print results
            if (palWord == revWord)
            {
                Console.WriteLine("");
                Console.WriteLine("{0} is a palindrome!", palWord);
            }
            else
            {
                Console.WriteLine("");
                Console.WriteLine("{0} is not a palindrome.", palWord);
            }
            Console.Read();
        }
    }
}

Longest & Shortest Word Finder

This is a C# console program that will accept a sentence and determine which word contains the largest amount of letters and which word contains the smallest amount of letters.

Longest & Shortest Word Finder C# Code

namespace Longest_Word
{
    class Program
    {
        static void Main(string[] args)
        {
            // Initialize the value
            string words = "";

            // Ask user to input a sentence
            Console.WriteLine("--- Longest and Shortest Word Finder ---");
            Console.WriteLine("");
            Console.WriteLine("This program will accept a sentence and determine which word contains the largest amount of letters and which word contains the smallest amount of letters.");
            Console.WriteLine("");
            Console.WriteLine("Type a sentence to find the longest and shortest words:");

            words = Console.ReadLine();

            // Break sentence apart into an array of strings
            string[] wordArray = words.Split(' ');

            // Sort the array by word length
            var sorted = wordArray.OrderBy(word => word.Length);

            // Create variables for the last (longest) and the first (shortest) words
            var longest = sorted.LastOrDefault();
            var shortest = sorted.FirstOrDefault();

            // Print the results
            Console.WriteLine("");
            Console.WriteLine("\"{0}\" is the longest word.", longest);
            Console.WriteLine("\"{0}\" is the shortest word.", shortest);
            Console.Read();
        }
    }
}

Word Filter

This is a C# console program that will accept a sentence and a number of letters, then it will list the words that are longer than the specified number of letters.

Word Filter C# Code

namespace Word_Filter
{
    class Program
    {
        static void Main(string[] args)
        {
            // Initialize the values
            string sentence = string.Empty;
            string[] words = { };
            string longWord = string.Empty;
            int minLimit = 0;
            string response = string.Empty;

            // Ask user to input a sentence
            Console.WriteLine("--- Word Filter ---");
            Console.WriteLine("");
            Console.WriteLine("Type a sentence:");

            sentence = Console.ReadLine();

            // Ask user for the minimum character number
            Console.WriteLine("");
            Console.WriteLine("This program will list the words longer than a certain number of letters.");
            Console.WriteLine("Enter the minimum number of letters:");
            bool result = Int32.TryParse(Console.ReadLine(), out minLimit);

            // Split sentence into individual words
            words = sentence.Split(' ');

            // Sort out the words that equal or exceed the minimum number of characters
            List sortedList = words.OrderByDescending(w => w.Length).ToList();
            List uniqueList = new List();

            foreach (string word in sortedList)
            {
                if (word.Length >= minLimit)
                {
                    var item = uniqueList.Find(x => x == word);
                    if (item == null)
                    {
                        uniqueList.Add(word);
                    }
                }
            }

            // Print results to console
            Console.WriteLine("");
            Console.WriteLine("The words that are longer than {0} letters are ", minLimit);
            foreach (string word in uniqueList)
            {
                Console.WriteLine(word);
            }
            Console.Read();
        }
    }
}

Word Frequency Finder

This is a C# console program that will accept a sentence and determine how many times each word appears in the sentence.

Word Frequency Finder C# Code

namespace Word_Frequency
{
    class Program
    {
        static void Main(string[] args)
        {
            // Initialize the values
            string sentence = string.Empty;
            string[] words = { };
            string longWord = string.Empty;
            int minLimit = 0;

            // Ask user to input a sentence
            Console.WriteLine("--- Word Frequency Finder ---");
            Console.WriteLine("");
            Console.WriteLine("This program will accept a sentence and determine how many times each word appears in the sentence.");
            Console.WriteLine("");
            Console.WriteLine("Type a sentence to see the frequency of each word:");

            sentence = Console.ReadLine();

            // Split sentence into individual words
            words = sentence.Split(' ');

            // Sort the list by number of characters
            List sortedList = words.OrderBy(w => w).ToList();

            // Counting list and Garbage list
            List freqList = new List();
            List usedList = new List();

            foreach (string word in sortedList)
            {
                freqList = sortedList.FindAll(x => x == word);

                // Ignore previously counted word
                var item = usedList.Find(x => x == word);
                if (item == null)
                {
                    usedList.Add(word);

                    // Print results
                    Console.WriteLine("\n{0} appears {1} time(s)", word, freqList.Count);
                }
            }
            Console.Read();
        }
    }
